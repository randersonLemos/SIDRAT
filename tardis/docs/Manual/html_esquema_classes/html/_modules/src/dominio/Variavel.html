[docs]class Variavel(Loggin):
    def __init__(self, parametro: Parametro, valor=None, posicao=None):
        super().__init__()

        self._name = __name__
        """
        Variável com o nome do arquivo
        """

        self._parametro = parametro
        """
        O parametro que representa essa variavel
        """

        self._valor = ""
        """
        Valor da variavel
        """

        self._posicao = -1
        """
        posicao da variavel no vector de domínio
        """
        v_valor, v_posicao = self._parametro.get_valor(valor)
        p_posicao, p_valor = self._parametro.get_posicao(posicao)
        if (v_valor == p_valor) and (v_posicao == p_posicao):
            self._valor = v_valor
            self._posicao = int(v_posicao)
        else:
            self._valor = v_valor
            self._posicao = int(v_posicao)
            self.log(tipo=EnumLogStatus.WARN, texto=f'A relação entre valor[{valor}] e posição[{posicao}] no domínio[{self._parametro.dominio}] esta errada. Assim foi gravado valor[{self._valor}] e posicao[{self._posicao}].')

    @property
    def nome(self):
        return self._parametro.nome

    @property
    def valor(self):
        return self._valor

    @property
    def posicao(self):
        return int(self._posicao)

    @property
    def parametro(self):
        return self._parametro

    @valor.setter
    def valor(self, valor):
        valor, posicao = self._parametro.get_valor(valor)
        self._valor = valor
        self._posicao = posicao

    @posicao.setter
    def posicao(self, posicao):
        if posicao < 0:
            self.log(texto=f'Não foi gravado nova posicao, pois é menor que zero.', tipo=EnumLogStatus.WARN)
            raise ValueError
        if posicao >= len(self._parametro.dominio):
            self.log(texto=f'Não foi gravado nova posicao da variavel [{self.nome}], pois é maior que dominio [{len(self._parametro.dominio)}].', tipo=EnumLogStatus.WARN)
            raise ValueError(f'Tamanho maximo {len(self._parametro.dominio)}')
        posicao, valor = self._parametro.get_posicao(posicao)
        self._valor = valor
        self._posicao = int(posicao)
