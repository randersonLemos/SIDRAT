[docs]    def add_in_estrategias(self, estrategia: Estrategia, sobrescreve: bool = False) -> bool:

        """
        Método que adiciona estrategia a lista, e verifica se já existe igual

        :param Estrategia estrategia: É a estrategia para ser adcionada
        :param bool sobrescreve: Se quando encontrar uma estrategia igual, deve sobrescrever. Padrão é não
        :return: Retorna True ou False, para verificar se a adição ocorreu corretamente.
        :rtype: bool
        """

        if estrategia is None:
            self.log(texto='Não pode add uma estrategia None.')
            return False

        identificador = copy.deepcopy(estrategia.id)
        iteracao = copy.deepcopy(estrategia.iteracao)

        if iteracao in self._estrategias:

            if len(self._estrategias[iteracao]) > 0:
                if identificador in self._estrategias[iteracao]:
                    if sobrescreve:
                        self._estrategias[iteracao][identificador] = copy.deepcopy(estrategia)
                        self.log(
                            texto=f'A estrategia com iteração [{iteracao}] e id [{identificador}] foi sobrescrita.')
                        return True
                    else:
                        self.log(tipo=EnumLogStatus.ERRO,
                                 texto=f'A Coleção já contem a estrategia com iteração [{iteracao}] e id [{identificador}]. Não foi adicionada')
                        return False
                else:
                    self._estrategias[iteracao][identificador] = estrategia
            else:
                self._estrategias[iteracao][identificador] = estrategia
        else:
            self._estrategias[iteracao] = {identificador: estrategia}
        return True



[docs]    def get_estrategias_by_iteracao(self, iteracao: [int], com_erro: bool = False) -> dict:

        """
        Método para obter as estrategias, atravez da iteracao.

        :param [int] iteracao: Número que identifica a iteração
        :param bool com_erro: Se igual False retorna todas sem erro.
        :return: Retorna as estrategias que atendem ao critério de busca
        :rtype: dict
        """

        try:
            aux = {}
            if iteracao is not None:
                if type(1) == type(iteracao):
                    iteracao = [iteracao]
                for it in iteracao:
                    if it in self._estrategias:
                       aux[it] = self._estrategias[it]

            retorno = {}
            if com_erro is None:
                retorno = aux
            else:
                for estrategias in aux.values():
                    for estrategia in estrategias.values():
                        if (com_erro is False) and ((estrategia.has_erro == "") or (estrategia.has_erro is None)):
                            if estrategia.iteracao not in retorno:
                                retorno[estrategia.iteracao] = {estrategia.id: estrategia}
                            else:
                                retorno[estrategia.iteracao][estrategia.id] = estrategia
                        elif (com_erro is True) and (not estrategia.has_erro == "") and (estrategia.has_erro is not None):
                            if estrategia.iteracao not in retorno:
                                retorno[estrategia.iteracao] = {estrategia.id: estrategia}
                            else:
                                retorno[estrategia.iteracao][estrategia.id] = estrategia

            if len(retorno) <= 0:
                self.log(tipo=EnumLogStatus.WARN, texto=f'Nenhuma estratégia com iteracao {iteracao} foi encontrada.')

            return retorno

        except Exception as ex:
            self.log(tipo=EnumLogStatus.ERRO, texto=f'Erro para obter estrategia', info_ex=str(ex))
            return {}

        """
        Converte as estrategias em um dataFrame. Seguindo a regra: Se identificar = None, retorna todas as estratégias da iteracao definida, Se iteração nao definida retorna todas as estratégias.

        TODO :param [int] iteracao: Número que identifica a iteração
        :param int identificador: Número do identificador
        TODO :param bool com_erro: Se igual a True retorna todas com erro, Se igual False retorna todas sem erro. Se igual None retorna tudo.
        :return: Retorna o data frame da opção selecionada
        :rtype: pd.DataFrame
        """

        # TODO fazer busca por lista de iteracao
        df = pd.DataFrame()
        if iteracao is not None:

            if iteracao in self._estrategias:
                if identificador is None:
                    for kEst, vEst in self._estrategias[iteracao].items():
                        df = pd.concat([df, vEst.data_frame()], axis=0)
                    return df
                else:
                    if identificador in self._estrategias[iteracao]:
                        return self._estrategias[iteracao][identificador].data_frame()
        else:
            for kIte, vIte in self._estrategias.items():
                for kEst, vEst in vIte.items():
                    df = pd.concat([df, vEst.data_frame()], axis=0)
            return df

        self.log(tipo=EnumLogStatus.WARN,
                 texto=f'Nenhuma estratégia com iteracao {iteracao} e id {identificador} foi encontrada.')
        return df

        serializacao = estrategia.serializacao()
        if len(self._estrategias) < 1:
            return False
        for estrategias_in in self._estrategias.values():
            for estrategia_in in estrategias_in.values():
                if serializacao == estrategia_in.serializacao():
                    return True
        return False

[docs]    def melhores_estrategias(self, quantidade=10):
        estrategias_of = []
        for it in list(map(int, self.estrategias)):
            for id in list(map(int, self.estrategias[it])):
                if (self.estrategias[it][id].of is not None) and (it >= 0):
                    estrategias_of.append({'it': it, 'id': id, 'of': self.estrategias[it][id].of})

        estrategias_of.sort(key=lambda x: x['of'], reverse=True)

        melhores_estrategias = Estrategias()
        for estrategia in estrategias_of[0:quantidade]:
            melhores_estrategias.add_in_estrategias(self.estrategias[estrategia['it']][estrategia['id']])

        return melhores_estrategias
