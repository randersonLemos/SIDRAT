[docs]    def set_contexto(self, contexto: Contexto) -> None:
        """
        Método de set o contexto para uso na classe

        :param Contexto contexto: Variavel responsavel por armazear toda informacao
        """
        if contexto is None:
            self.log(tipo=EnumLogStatus.ERRO_FATAL, texto='Erro ao setar contexto, contexto esta None.')

        self._contexto = contexto


    @property
    def necessidade(self):
        return self._necessidade

    @necessidade.setter
    def necessidade(self, necessidade):
        self._necessidade = self._necessidade + necessidade

    def _check_necessidades(self):
        """
        Verificar se a lista em self._list esta presente no contexto
        """

        tem_tudo = True
        msg = ""
        for need in self.necessidade:
            if self._contexto.tem_atributo(need):
                self.log(texto=f"Atritubo [{need}] definico como [{self._contexto.get_atributo(need)}]")
            else:
                tem_tudo = False
                msg = f'Atributo [{need}] não foi definido\n{msg}'

        if not tem_tudo:
            self.log(tipo=EnumLogStatus.ERRO_FATAL, texto=msg)
