[docs]    def run(self, contexto: Contexto) -> Contexto:
        """
        Executa a otimizacao usando o método MCC.

        :param Contexto contexto: contexto com todas as informações necessárias
        :return: Devolve o contexto atualizado
        :rtype: Contexto
        """

        super().run(contexto)
        self._check_necessidades()
        self._qtd_ocorrencia_sem_melhora = self._contexto.get_atributo(EnumAtributo.OTIMIZACAO_MCC_ELIMINA_NIVEIS_QUANTIDADE)

        self._estrategia_base: Estrategia = self._contexto.get_atributo(EnumAtributo.ESTRATEGIA_BASE)
        self._estrategias = self._contexto.get_atributo(EnumAtributo.ESTRATEGIAS)
        self._estrategia_melhor = self._estrategias.melhor_estrategia()
        if self._estrategia_melhor.id == 0:
            self._estrategia_melhor = copy.deepcopy(self._estrategia_base)
        self._iteracao = max(list(self._estrategias.estrategias))
        self._id = max(list(self._estrategias.estrategias[self._iteracao]))
        self._variaveis_por_iteracao = self._contexto.get_atributo(EnumAtributo.OTIMIZACAO_MCC_VARIAVEIS_ITERACAO)
        variaveis = self._estrategia_melhor.get_variavies_by_tipo()

        for nome in variaveis:
            self._qtd_niveis_totais += len(variaveis[nome].parametro.dominio)
            self._dic_niveis_variaveis_sem_melhora[nome] = {}
            self._qtd_variaveis += 1

        if self._contexto.tem_atributo(EnumAtributo.OTIMIZACAO_MCC_DIC_NIVEIS_VARIAVEIS_SEM_MELHORA):
            self._dic_niveis_variaveis_sem_melhora = self._contexto.get_atributo(EnumAtributo.OTIMIZACAO_MCC_DIC_NIVEIS_VARIAVEIS_SEM_MELHORA)

        avaliacao = self._contexto.get_modulo(EnumModulo.AVALIACAO)
        exportacao = Exportacao()

        criterio = self._contexto.get_modulo(EnumModulo.CRITERIOPARADA)
        while not criterio.run(self._contexto):
            self._iteracao += 1

            if self._proximo_conjuto():
                self._contexto.set_atributo(EnumAtributo.ESTRATEGIAS, self._estrategias, True)
                self._contexto.set_atributo(EnumAtributo.AVALIACAO_ITERACAO_AVALIAR, [self._iteracao], True)

                self._contexto = avaliacao.run(self._contexto)
                self._estrategia_melhor = self._contexto.get_atributo(EnumAtributo.ESTRATEGIAS).melhor_estrategia()

                for ii in self._estrategias.estrategias[self._iteracao]:
                    try:
                        estrategia = self._estrategias.estrategias[self._iteracao][ii]
                        nome, valor_old, valor_new, vazio = estrategia.geral.replace(":", "").replace(" ", "").replace("->","").replace("[","").split("]")

                        if estrategia.geral != self._estrategia_melhor.geral:
                            if valor_new in self._dic_niveis_variaveis_sem_melhora[nome]:
                                self._dic_niveis_variaveis_sem_melhora[nome][str(valor_new)] += 1
                            else:
                                self._dic_niveis_variaveis_sem_melhora[nome][str(valor_new)] = 1
                        else:
                            self._dic_niveis_variaveis_sem_melhora[nome][str(valor_new)] = 0

                    except Exception as ex:
                        self.log(tipo=EnumLogStatus.WARN, texto=f"Não foi possivel fazer o parser de [{estrategia.geral}]")

                self._contexto.set_atributo(EnumAtributo.OTIMIZACAO_MCC_DIC_NIVEIS_VARIAVEIS_SEM_MELHORA, [self._dic_niveis_variaveis_sem_melhora], True)
                exportacao.csv(self._contexto)
                exportacao.obejto(self._contexto)
            else:
                self.log(texto=f'Fim da execução do {__name__}')
                break

        return self._contexto


    def _proximo_conjuto(self) -> bool:
        variaveis_avaliadas = 0
        self.log(texto=f'Buscar estrategias para iteracao {self._iteracao}.')
        variaveis = copy.deepcopy(self._estrategia_melhor.variaveis)
        qtd_variaveis = len(variaveis.get_variaveis_by_tipo())
        qtd_tentiva_achar_varaivel = 0
        qtd_variaveis_descartadas = 0

        while self._variaveis_por_iteracao >= variaveis_avaliadas:
            qtd_tentiva_achar_varaivel += 1

            while len(self._variaveis_nome) < self._variaveis_por_iteracao:
                self._variaveis_nome = list(self._variaveis_nome) + list(self._estrategia_melhor.get_variavies_by_tipo())

            variavel_nome = self._variaveis_nome[0]
            #definindo o passo sempre ocmo 1 ou -1 pr dentro do metodo ele percorrer os passos.
            estrategia_mais_um:  Estrategia = self._evoluindo_variavel(+1, variaveis.get_variavel_by_nome(variavel_nome))
            estrategia_menos_um: Estrategia = self._evoluindo_variavel(-1, variaveis.get_variavel_by_nome(variavel_nome))

            if (estrategia_mais_um is not None) or (estrategia_menos_um is not None):
                self._estrategias.add_in_estrategias(estrategia_mais_um)
                self._estrategias.add_in_estrategias(estrategia_menos_um)
                variaveis_avaliadas += 1
                qtd_tentiva_achar_varaivel = 0
            else:
                qtd_variaveis_descartadas += 1

            del self._variaveis_nome[0]

            if (qtd_tentiva_achar_varaivel > qtd_variaveis) or (qtd_variaveis_descartadas > self._qtd_niveis_totais):
                if variaveis_avaliadas > 0:
                    return True
                else:
                    self.log(texto=f'Convergiu por não encontrar caminho para seguir.')
                    return False
        return True

    def _evoluindo_variavel(self, passo: int, variavel: Variavel) -> Estrategia:

        estrategia_aux = None

        while True:
            variavel = copy.deepcopy(variavel)

            if abs(passo) > len(variavel.parametro.dominio):
                estrategia_aux = None
                self.log(texto=f'Não existe dominio para mais um na variavel [{variavel.nome}], com valor [{variavel.valor}].')
                break
            elif variavel.posicao + passo < 0:
                estrategia_aux = None
                self.log(texto=f'Não existe dominio para mais um na variavel [{variavel.nome}], com valor [{variavel.valor}].')
                break
            elif variavel.posicao + passo >= len(variavel.parametro.dominio):
                estrategia_aux = None
                self.log(texto=f'Não existe dominio para mais um na variavel [{variavel.nome}], com valor [{variavel.valor}].')
                break

            valor_antigo = variavel.valor
            try:
                variavel.posicao += passo
            except ValueError as ex:
                self.log(texto="Valor for do limite", info_ex=str(ex))
                return None

            # verificando se o nivel já foi usado varias vezes
            if self._contexto.get_atributo(EnumAtributo.OTIMIZACAO_MCC_ELIMINA_NIVEIS):
                if (str(variavel.valor) in self._dic_niveis_variaveis_sem_melhora[variavel.nome]) and (self._dic_niveis_variaveis_sem_melhora[variavel.nome][str(variavel.valor)] >= self._qtd_ocorrencia_sem_melhora):
                    self.log(tipo=EnumLogStatus.WARN, texto=f'Variavel [{variavel.nome}] no nível [{variavel.valor}] não é usada mais')
                    return None

            self.log(texto=f'Passo igual [{passo}] para evolução da variavel [{variavel.nome}], valor antigo [{valor_antigo}], valor novo [{variavel.valor}]')
            estrategia_aux = Estrategia(iteracao=self._iteracao, id=self._id + 1, estrategia=self._estrategia_melhor)
            estrategia_aux.variaveis.set_variavel_posicao_by_nome(variavel.nome, variavel.posicao)
            estrategia_aux.geral = f'[{variavel.nome}]: [{valor_antigo}] -> [{variavel.valor}]'

            if self._estrategias.existe_estrategia(estrategia_aux):
                passo += int(passo / abs(passo))
                self.log(texto=f'Já existe estrategia com esses valores nas variaveis. Passo será [{passo}].')
            else:
                self._id += 1
                break

        return estrategia_aux
