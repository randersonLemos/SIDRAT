[docs]    def run(self, contexto: Contexto) -> Contexto:
        """
        Executa o PSO

        :param Contexto contexto: contexto com todas as informações necessárias
        :return: Devolve o contexto atualizado
        :rtype: Contexto
        """

        super().run(contexto)
        self._check_necessidades()
        np.random.seed(1)

        self._estrategia_base: Estrategia = self._contexto.get_atributo(EnumAtributo.ESTRATEGIA_BASE)
        self._variaveis = self._estrategia_base.get_variavies_by_tipo()
        self._estrategias = self._contexto.get_atributo(EnumAtributo.ESTRATEGIAS)
        self._iteracao = max(list(self._estrategias.estrategias))
        self._id = max(list(self._estrategias.estrategias[self._iteracao]))
        self._quantidade_populacao = self._contexto.get_atributo(EnumAtributo.OTIMIZACAO_PSO_QTD_POPULACAO)
        avaliacao = self._contexto.get_modulo(EnumModulo.AVALIACAO)
        exportacao = Exportacao()

        self._inicializa_populacao()

        criterio = self._contexto.get_modulo(EnumModulo.CRITERIOPARADA)
        while not criterio.run(self._contexto):

            self._gerar_estrategias_avalia(avaliacao)

            for ii in range(self._quantidade_populacao):
                # if self._populacao[ii]['id'] < 0:
                #     continue
                rp = np.random.rand(1)
                rg = np.random.rand(1)
                ld = np.zeros(len(self._variaveis))
                lu = np.zeros(len(self._variaveis))
                for jj in range(len(self._variaveis)):
                    ld[jj] = 0
                    lu[jj] = self._populacao[ii]['variavel'][jj]['lu']

                self._populacao[ii]['velocidade'] = self._w * self._populacao[ii]['velocidade'] + \
                                                  self._cp * rp * (self._populacao[ii]['best_posicao'] - self._populacao[ii]['posicao']) + \
                                                  self._cg * rg * (self._gbest['posicao'] - self._populacao[ii]['posicao'])

                posicao_atualizada = self._populacao[ii]['posicao'] + self._populacao[ii]['velocidade']
                posicao_atualizada = np.maximum(posicao_atualizada, ld)
                posicao_atualizada = np.minimum(posicao_atualizada, lu)
                self._populacao[ii]['posicao'] = posicao_atualizada.astype(int)

                for vv in range(len(self._populacao[ii]['variavel'])):
                    self._populacao[ii]['variavel'][vv]['posicao'] = self._populacao[ii]['posicao'][vv]

            self._w *= self._wdamp

            self._contexto.set_atributo(EnumAtributo.OTIMIZACAO_PSO_POPULACAO, self._populacao, True)
            self._contexto.set_atributo(EnumAtributo.OTIMIZACAO_PSO_W, self._w, True)

            exportacao.csv(self._contexto)
            exportacao.obejto(self._contexto)

        return self._contexto


    def _inicializa_populacao(self):
        empty_particle = {
            'posicao': None,
            'velocidade': None,
            'of': None,
            'best_posicao': None,
            'best_of': np.inf,
            'variavel': None,
            'id': None,
        }

        if self._contexto.tem_atributo(EnumAtributo.OTIMIZACAO_PSO_POPULACAO):
            self._populacao = self._contexto.get_atributo(EnumAtributo.OTIMIZACAO_PSO_POPULACAO)

            if self._contexto.tem_atributo(EnumAtributo.OTIMIZACAO_PSO_W):
                self._w = self._contexto.get_atributo(EnumAtributo.OTIMIZACAO_PSO_W)
            return

        self._populacao = []
        for ii in range(self._quantidade_populacao):
            self._populacao.append(empty_particle.copy())
            self._populacao[ii]['of'] = float('inf')  # ofFunction(self._populacao[ii]['posicao'])
            self._populacao[ii]['velocidade'] = np.zeros(len(self._variaveis))
            self._populacao[ii]['posicao'] = np.zeros(len(self._variaveis))
            self._populacao[ii]['variavel'] = [None] * (len(self._variaveis))

            pp = 0
            for nome in self._variaveis:
                lu = len(self._variaveis[nome].parametro.dominio)-1
                rand_posicao = np.random.randint(0, high=lu+1, size=1)

                self._populacao[ii]['posicao'][pp] = int(rand_posicao[0])
                self._populacao[ii]['variavel'][pp] = {'nome': nome, 'lu': lu, 'posicao': int(rand_posicao[0])}

                pp += 1

            self._populacao[ii]['best_posicao'] = deepcopy(self._populacao[ii]['posicao'])
            self._populacao[ii]['best_of'] = deepcopy(self._populacao[ii]['of'])

    def _gerar_estrategias_avalia(self, avaliacao):
        self._iteracao += 1
        for ii in range(self._quantidade_populacao):
            self._id += 1
            self._populacao[ii]['id'] = self._id
            estrategia = Estrategia(id=self._id, iteracao=self._iteracao, estrategia=self._estrategia_base)

            for jj in range(len(self._populacao[ii]['posicao'])):
                nome = self._populacao[ii]['variavel'][jj]['nome']
                posicao = self._populacao[ii]['variavel'][jj]['posicao']
                lu = self._populacao[ii]['variavel'][jj]['lu']
                if int(lu) < int(posicao):
                    self.log(tipo=EnumLogStatus.ERRO, texto=f"Posicao da varivael {nome} esta fora dos limites.")
                if int(posicao) != int(self._populacao[ii]['posicao'][jj]):
                    self.log(tipo=EnumLogStatus.ERRO, texto="Erro posicoes diferentes")
                try:
                    estrategia.variaveis.set_variavel_posicao_by_nome(nome, posicao)
                except Exception as ex:
                    self.log(tipo=EnumLogStatus.ERRO, texto="Erro ao setar variavel", info_ex=str(ex))
                    self._populacao[ii]['posicao'][jj] = 0
                    estrategia.variaveis.set_variavel_posicao_by_nome(nome, 0)

            if self._estrategias.existe_estrategia(estrategia):
                self._populacao[ii]['id'] = -1
                self._id -= 1
            else:
                self._estrategias.add_in_estrategias(estrategia)

        self._contexto.set_atributo(EnumAtributo.ESTRATEGIAS, self._estrategias, True)
        self._contexto.set_atributo(EnumAtributo.AVALIACAO_ITERACAO_AVALIAR, [self._iteracao], True)
        self._contexto = avaliacao.run(self._contexto)
        self._estrategias = self._contexto.get_atributo(EnumAtributo.ESTRATEGIAS)

        for ii in range(len(self._populacao)):
            id = self._populacao[ii]['id']
            if id < 0:
                continue

            estrategia: Estrategia = self._estrategias.get_estrategia_by_iteracao_id(self._iteracao, id)

            if (estrategia.of is None) or (estrategia.has_erro is not None):
                self._populacao[ii]['of'] = float('inf')
            else:
                self._populacao[ii]['of'] = -1 * estrategia.of

                if self._populacao[ii]['of'] <= self._populacao[ii]['best_of']:
                    self.log(texto=f"\t\t+[LOCAL] [{self._iteracao:3}][{id:4}]: of {self._populacao[ii]['best_of']:3} > {self._populacao[ii]['of']:15}")
                    self._populacao[ii]['best_posicao'] = deepcopy(self._populacao[ii]['posicao'])
                    self._populacao[ii]['best_of'] = deepcopy(self._populacao[ii]['of'])

                    if self._populacao[ii]['of'] < self._gbest['of']:
                        self.log(texto=f"\t+[GLOBAL] [{self._iteracao:3}][{id:4}]: of {self._gbest['of']:3} > {self._populacao[ii]['of']:15}")
                        self._gbest['posicao'] = deepcopy(self._populacao[ii]['posicao'])
                        self._gbest['variavel'] = deepcopy(self._populacao[ii]['variavel'])
                        self._gbest['of'] = deepcopy(self._populacao[ii]['of'])
